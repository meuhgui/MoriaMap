package dev.moriamap.model;

/**
 * Represents instants of time with second precision.
 */
public class Time {

    /**
     * Number of hours of this Time.
     */
    private int hours;

    /**
     * Number of minutes of this Time.
     */
    private int minutes;

    /**
     * Number of second of this Time.
     */
    private int seconds;

    /**
     * Class constructor specifying hours, minutes and seconds.
     * @param hours the amount of hours
     * @param minutes the amount of minutes
     * @param seconds the amount of seconds
     */
    public Time(int hours, int minutes, int seconds) {
        if (hours < 0 || minutes < 0)
            throw new IllegalArgumentException("Negative argument");
        if (minutes > 59) {
            hours += minutes / 60;
            minutes %= 60;
        }
        this.hours = hours;
        this.minutes = minutes;
    }

    /**
     * {@return the hours of this Time}
     */
    public int getHours() {
        return this.hours;
    }

    /**
     * {@return the minutes of this Time}
     */
    public int getMinutes() {
        return this.minutes;
    }

    /**
     * {@return the seconds of this Time}
     */
    public int getSeconds() {
        return this.seconds;
    }
}
